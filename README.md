while True :
  kode = input("input")

  match kode :
      case"A01":
         print ("nama barang : snack")
         print ("harga : 10000")
      case"A02":
         print ("nama barang : minuman")
         print ("harga : 8000")
      case"B01":
         print ("nama barang : mie instan")
         print ("harga : 3000")
      case"B02":
         print ("nama barang : sabun")
         print ("harga : 5000")
      case"C01":
         print ("nama barang : shampoo")
         print ("harga : 12000")
